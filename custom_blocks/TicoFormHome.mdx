  <div className="tico-form-wrapper">
    <form
      className="tico-form"
      onSubmit={(e) => {
        e.preventDefault();

        const input = e.target.elements['tico-message'];
        const message = input?.value?.trim() || null;

        if (message) {
          window.cleanStartTico();
          setTimeout(() => {
            window.searchTico(message);
          }, 1500);
          input.value = '';
        } else if (!message) {
          console.warn('Message is empty');
        } else {
          console.warn('Voiceflow chat is not loaded');
        }

        return false;
      }}
    >
      <div className="tico-input-wrapper">
        <input
          className="tico-input"
          type="text"
          id="tico-message"
          name="tico-message"
          placeholder="Hey Tico! I need help with..."
        />
        <button className="tico-submit" type="submit">
          <p>Send</p>
        </button>
      </div>
      <img
        className="tico-avatar-secondary"
        src="https://files.readme.io/fd68b8868d6a64767d905db825614801e75313602a00b7d664618dec2b22c153-Frame_48095776.png"
