import { useState, useEffect, useRef } from 'react';

export function Video({ src }) {
  const [isLoaded, setIsLoaded] = useState(false);
  const videoRef = useRef(null);

  useEffect(() => {
    const video = videoRef.current;
    if (!video) return;

    // Reset on src change
    setIsLoaded(false);

    // If the video is already buffered enough, mark loaded immediately
    // 2 = HAVE_CURRENT_DATA, 4 = HAVE_ENOUGH_DATA
    if (video.readyState >= 2) {
      setIsLoaded(true);
      return;
    }

    const markLoaded = () => setIsLoaded(true);
    const onError = (e) => {
      // Surface errors during dev; remove or customize for prod
      console.error('Video load error:', e);
      // Keep placeholder visible if error occurs
    };

    video.addEventListener('loadeddata', markLoaded, { once: true });
    video.addEventListener('canplay', markLoaded, { once: true });
    video.addEventListener('playing', markLoaded, { once: true });
    video.addEventListener('loadedmetadata', () => {
      // Some browsers need a nudge to start; attempt play silently
      // If it fails due to autoplay policy, muted+playsInline should help.
      video.play?.().catch(() => {});
    }, { once: true });
    video.addEventListener('error', onError);

    return () => {
      video.removeEventListener('loadeddata', markLoaded);
      video.removeEventListener('canplay', markLoaded);
      video.removeEventListener('playing', markLoaded);
      video.removeEventListener('loadedmetadata', markLoaded);
      video.removeEventListener('error', onError);
    };
  }, [src]);

  return (
    <div
      style={{
        width: '100%',
        margin: '2rem auto',
        borderRadius: '8px',
        backgroundColor: 'black',
        position: 'relative',
        overflow: 'hidden',
        // Reserve a 16:9 box only until loaded
        paddingBottom: isLoaded ? 0 : '56.25%',
      }}
    >
      {/* Video stays mounted; we fade it in to avoid layout jumps */}
      <video
        ref={videoRef}
        src={src}
        autoPlay
        muted
        loop
        playsInline
        preload="auto"
        style={{
          // Before load: cover the placeholder 16:9 box
          // After load: use natural aspect ratio (height:auto)
          position: isLoaded ? 'static' : 'absolute',
          top: isLoaded ? undefined : 0,
          left: isLoaded ? undefined : 0,
          width: '100%',
          height: isLoaded ? 'auto' : '100%',
          objectFit: isLoaded ? 'contain' : 'cover',
          borderRadius: '8px',
          opacity: isLoaded ? 1 : 0,
          transition: 'opacity 0.3s ease-in-out',
        }}
      />
    </div>
  );
}

export default Video;
