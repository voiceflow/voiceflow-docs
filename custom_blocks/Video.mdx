import { useState, useEffect, useRef } from 'react';

export function Video({ src }) {
  const [ready, setReady] = useState(false);
  const videoRef = useRef(null);

  useEffect(() => {
    const v = videoRef.current;
    if (!v) return;

    // Reset whenever src changes
    setReady(false);

    // If metadata or frame is already available, flip immediately
    if (v.readyState >= 2) {
      setReady(true);
      return;
    }

    const markReady = () => setReady(true);
    const onError = (e) => {
      console.error('Video load error:', e);
      // Keep placeholder; don't flip to ready on error
    };

    v.addEventListener('loadedmetadata', markReady, { once: true });
    v.addEventListener('loadeddata', markReady, { once: true });
    v.addEventListener('canplay', markReady, { once: true });
    v.addEventListener('error', onError);

    return () => {
      v.removeEventListener('loadedmetadata', markReady);
      v.removeEventListener('loadeddata', markReady);
      v.removeEventListener('canplay', markReady);
      v.removeEventListener('error', onError);
    };
  }, [src]);

  return (
    <div style={{ width: '100%', margin: '2rem auto' }}>
      {/* Placeholder exists ONLY while not ready */}
      {!ready && (
        <div
          style={{
            width: '100%',
            background: 'black',
            borderRadius: '8px',
            // classic 16:9 aspect-ratio box
            position: 'relative',
          }}
        >
          <div
            style={{
              width: '100%',
              paddingBottom: '56.25%', // 16:9
            }}
          />
        </div>
      )}

      <video
        ref={videoRef}
        src={src}
        autoPlay
        muted
        loop
        playsInline
        preload="auto"
        // No display:none toggling—just fade for niceness
        style={{
          display: 'block',
          width: '100%',
          height: 'auto',          // natural aspect, no cutoff
          borderRadius: '8px',
          opacity: ready ? 1 : 0,  // won't affect layout
          transition: 'opacity 200ms ease-in-out',
        }}
        onLoadedData={() => setReady(true)}
      />
    </div>
  );
}

export default Video;
