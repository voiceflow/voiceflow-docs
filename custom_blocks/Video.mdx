import { useState, useEffect, useRef } from 'react';

export function Video({ src }) {
  const [isLoaded, setIsLoaded] = useState(false);
  const videoRef = useRef(null);

  useEffect(() => {
    const v = videoRef.current;
    if (!v) return;

    // Reset when src changes
    setIsLoaded(false);

    // If already buffered enough, mark loaded immediately
    if (v.readyState >= 2) {
      setIsLoaded(true);
      return;
    }

    const markLoaded = () => setIsLoaded(true);
    const onError = (e) => console.error('Video load error:', e);

    v.addEventListener('loadeddata', markLoaded, { once: true });
    v.addEventListener('canplay', markLoaded, { once: true });
    v.addEventListener('playing', markLoaded, { once: true });
    v.addEventListener('error', onError);

    return () => {
      v.removeEventListener('loadeddata', markLoaded);
      v.removeEventListener('canplay', markLoaded);
      v.removeEventListener('playing', markLoaded);
      v.removeEventListener('error', onError);
    };
  }, [src]);

  return (
    <div
      style={{
        width: '100%',
        margin: '2rem auto',
        borderRadius: '8px',
        backgroundColor: 'black',
        overflow: 'hidden',
      }}
    >
      {/* 16:9 placeholder only while loading */}
      {!isLoaded && (
        <div
          style={{
            width: '100%',
            paddingBottom: '56.25%', // 16:9
            backgroundColor: 'black',
            borderRadius: '8px',
          }}
        />
      )}

      <video
        ref={videoRef}
        src={src}
        autoPlay
        muted
        loop
        playsInline
        preload="auto"
        onLoadedData={() => setIsLoaded(true)}
        style={{
          display: 'block',
          width: '100%',
          height: 'auto',            // natural aspect ratio, no letterboxing
          borderRadius: '8px',
          opacity: isLoaded ? 1 : 0, // fade in
          transition: 'opacity 200ms ease-in-out',
        }}
      />
    </div>
  );
}

export default Video;
