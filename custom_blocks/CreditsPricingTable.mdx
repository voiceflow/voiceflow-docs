import React, { useEffect, useState } from 'react';

export function CreditsPricingTable() {
  const displayNames = {
    orchestration: "Orchestration",
    llm: "LLMs (Large Language Models)",
    asr: "STT (Speech-to-Text)",
    tts: "TTS (Text-to-Speech)"
  };

  const vendorMap = {
    openai: 'OpenAI',
    'open-ai': 'OpenAI',
    anthropic: 'Anthropic',
    google: 'Google',
    groq: 'Groq',
    deepgram: 'Deepgram',
    rimelabs: 'Rime Labs',
    cartesia: 'Cartesia',
    elevenlabs: 'ElevenLabs',
    'eleven-labs': 'ElevenLabs',
    assemblyai: 'AssemblyAI'
  };

  const getItemName = (item, vendor, category, io) => {
    if (category === 'orchestration') {
      if (item === 'interaction') return 'Chat usage';
      if (item === 'audio') return 'Voice usage';
      if (item === 'no-reply') return 'No-reply usage';
    }
    if (category === 'llm' && io) return `${item} (${io})`;
    if (category === 'asr') {
      if (vendor && vendor.toLowerCase().includes('deepgram')) return 'Deepgram - STT';
      if (vendor && vendor.toLowerCase().includes('assemblyai')) return 'Assemblyai - STT';
      if (vendor && vendor.toLowerCase().includes('cartesia')) return 'Cartesia - STT';
    }
    if (category === 'tts') {
      if (vendor && vendor.toLowerCase().includes('deepgram')) return 'Deepgram - TTS';
      if (vendor && vendor.toLowerCase().includes('rimelabs')) return 'Rime Labs - TTS';
      if (vendor && vendor.toLowerCase().includes('cartesia')) return 'Cartesia - TTS';
      if (vendor && vendor.toLowerCase().includes('elevenlabs')) {
        if (item && item.toLowerCase().includes('flash')) return 'Eleven Labs - TTS - Flash';
        if (item && item.toLowerCase().includes('turbo')) return 'Eleven Labs - TTS - Turbo';
        if (item && item.toLowerCase().includes('multilingual')) return 'Eleven Labs - TTS - Multilingual';
        return 'Eleven Labs - TTS';
      }
    }
    return item;
  };

  const getUnit = (unit) => {
    if (!unit) return '';
    const u = unit.toLowerCase();
    if (u.includes('per 1000 input tokens')) return '1K Input Tokens';
    if (u.includes('per 1000 output tokens')) return '1K Output Tokens';
    if (u.includes('credits per 1000 llm tokens')) return '1K Tokens';
    if (u.includes('credits per 1000 characters')) return '1K Characters';
    if (u.includes('credits per minute')) return 'Minutes';
    return unit;
  };

  const formatCredits = (value, priorityValue) => {
    if (value == null) return '';
    if (priorityValue != null && priorityValue !== '') {
      return (
        <>
          {value}
          <br />
          <span style={{ color: '#bdbdbd', fontSize: '0.85em' }}>
            {priorityValue} with priority processing
          </span>
        </>
      );
    }
    return String(value);
  };

  const [creditsData, setCreditsData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    async function fetchCreditsData() {
      setLoading(true);
      setError(null);
      try {
        const res = await fetch('https://realtime-api.voiceflow.com/v1alpha1/estimation/config');
        if (!res.ok) throw new Error('Failed to fetch pricing data');
        const data = await res.json();

        const grouped = {};
        const normalizeCategory = (c) => (c && c.startsWith('llm') ? 'llm' : c);

        for (const item of data.config.usage) {
          const io =
            item.category === 'llm-input' ? 'input' :
            item.category === 'llm-output' ? 'output' : undefined;

          const cat = normalizeCategory(item.category);
          if (!grouped[cat]) grouped[cat] = [];
          grouped[cat].push({ ...item, io });
        }

        setCreditsData(grouped);
      } catch (e) {
        setError(e.message);
      } finally {
        setLoading(false);
      }
    }
    fetchCreditsData();
  }, []);

  if (loading) return <div>Loading pricing data...</div>;
  if (error) return <div style={{ color: 'red' }}>Error: {error}</div>;
  if (!creditsData) return null;

  const getVendorName = (vendor) => {
    if (!vendor) return '';
    const key = vendor.toLowerCase().replace(/\s+/g, '').replace(/-/g, '');
    return vendorMap[key] || vendorMap[vendor.toLowerCase()] || vendor;
  };

  const interactionItem = creditsData.orchestration?.find(i => i.item === 'interaction');
  const noReplyItem = creditsData.orchestration?.find(i => i.item === 'no-reply');
  const audioItem = creditsData.orchestration?.find(i => i.item === 'audio');

  return (
    <div>
      <div style={{ marginBottom: '2rem' }}>
        <h2 style={{ fontWeight: 600, marginBottom: '1rem' }}>{displayNames.orchestration}</h2>
        <div className="rdmd-table">
          <div className="rdmd-table-inner">
            <table>
              <thead>
                <tr>
                  <th style={{ textAlign: 'left' }}>Vendor</th>
                  <th style={{ textAlign: 'left' }}>Item</th>
                  <th style={{ textAlign: 'left' }}>Unit</th>
                  <th style={{ textAlign: 'left' }}>Credits per Unit</th>
                </tr>
              </thead>
              <tbody>
                <tr>
                  <td style={{ textAlign: 'left' }}>Voiceflow</td>
                  <td style={{ textAlign: 'left' }}>Chat usage</td>
                  <td style={{ textAlign: 'left' }}>
                    Every time a conversation begins, the user sends a message, or a request is sent to the Dialogue Manager API.
                  </td>
                  <td style={{ textAlign: 'left' }}>
                    {interactionItem
                      ? formatCredits(interactionItem.value, interactionItem.priorityValue)
                      : '1'}
                  </td>
                </tr>
                <tr>
                  <td style={{ textAlign: 'left' }}>Voiceflow</td>
                  <td style={{ textAlign: 'left' }}>No-reply usage</td>
                  <td style={{ textAlign: 'left' }}>Every time a message is sent by the no-reply feature</td>
                  <td style={{ textAlign: 'left' }}>
                    {noReplyItem
                      ? formatCredits(noReplyItem.value, noReplyItem.priorityValue)
                      : '1'}
                  </td>
                </tr>
                <tr>
                  <td style={{ textAlign: 'left' }}>Voiceflow</td>
                  <td style={{ textAlign: 'left' }}>Voice usage</td>
                  <td style={{ textAlign: 'left' }}>Minutes of Voice AI</td>
                  <td style={{ textAlign: 'left' }}>
                    {audioItem
                      ? formatCredits(audioItem.value, audioItem.priorityValue)
                      : '10'}
                  </td>
                </tr>
              </tbody>
            </table>
          </div>
        </div>
      </div>

      {Object.entries(displayNames).filter(([type]) => type !== 'orchestration').map(([type, display]) => {
        let items = creditsData[type];
        if (!items || items.length === 0) return null;

        items = [...items].sort((a, b) => {
          if (['llm', 'asr', 'tts'].includes(type)) {
            const vendorCompare = getVendorName(a.vendor).localeCompare(getVendorName(b.vendor));
            if (vendorCompare !== 0) return vendorCompare;
            const nameA = getItemName(a.item, a.vendor, type, a.io);
            const nameB = getItemName(b.item, b.vendor, type, b.io);
            const nameCompare = nameA.localeCompare(nameB);
            if (nameCompare !== 0) return nameCompare;
            return (a.value ?? Infinity) - (b.value ?? Infinity);
          }
          return getItemName(a.item, a.vendor, type, a.io).localeCompare(
            getItemName(b.item, b.vendor, type, b.io)
          );
        });

        const isLLM = type === 'llm';

        let combinedItems = items;
        if (isLLM) {
          const map = {};
          for (const it of items) {
            const key = `${it.vendor}||${it.item}`;
            if (!map[key]) map[key] = { vendor: it.vendor, item: it.item, input: null, output: null };
            if (it.io === 'input') map[key].input = it;
            if (it.io === 'output') map[key].output = it;
          }
          combinedItems = Object.values(map);
        }

        return (
          <div key={type} style={{ marginBottom: '2rem' }}>
            <h2 style={{ fontWeight: 600, marginBottom: '1rem' }}>{display}</h2>
            <div className="rdmd-table">
              <div className="rdmd-table-inner">
                <table>
                  <thead>
                    <tr>
                      <th style={{ textAlign: 'left' }}>Vendor</th>
                      <th style={{ textAlign: 'left' }}>Item</th>
                      {isLLM ? (
                        <>
                          <th style={{ textAlign: 'left' }}>Cost per 1K output tokens</th>
                          <th style={{ textAlign: 'left' }}>Cost per 1K input tokens</th>
                        </>
                      ) : (
                        <>
                          <th style={{ textAlign: 'left' }}>Unit</th>
                          <th style={{ textAlign: 'left' }}>Credits per Unit</th>
                        </>
                      )}
                    </tr>
                  </thead>
                  <tbody>
                    {(isLLM ? combinedItems : items).map((item, index) => (
                      <tr key={`${item.vendor}-${item.item}-${item.io}-${index}`}>
                        <td style={{ textAlign: 'left' }}>{getVendorName(item.vendor)}</td>
                        <td style={{ textAlign: 'left' }}>{getItemName(item.item, item.vendor, type, item.io)}</td>

                        {isLLM ? (
                          <>
                            <td style={{ textAlign: 'left' }}>
                              {item.output ? formatCredits(item.output.value, item.output.priorityValue) : ''}
                            </td>
                            <td style={{ textAlign: 'left' }}>
                              {item.input ? formatCredits(item.input.value, item.input.priorityValue) : ''}
                            </td>
                          </>
                        ) : (
                          <>
                            <td style={{ textAlign: 'left' }}>{getUnit(item.unit)}</td>
                            <td style={{ textAlign: 'left' }}>{formatCredits(item.value, item.priorityValue)}</td>
                          </>
                        )}
                      </tr>
                    ))}
                  </tbody>
                </table>
              </div>
            </div>
          </div>
        );
      })}
    </div>
  );
}

<CreditsPricingTable />}
