import { useState, useEffect } from 'react';

export const CreditCalculator = () => {
  // THIS IS PRETTY MUCH ENTIRELY GENERATED WITH CURSOR
  // I DON'T RECOMMEND MODIFYING IT MANUALLY IF YOU CAN AVOID IT
	// 


  // State for API data
  const [creditsData, setCreditsData] = useState(null);
  const [plans, setPlans] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  // State for user inputs
  const [voiceMinutes, setVoiceMinutes] = useState(500);
  const [chatMessages, setChatMessages] = useState(1000);
  const [voiceMinutesInput, setVoiceMinutesInput] = useState("500");
  const [chatMessagesInput, setChatMessagesInput] = useState("1000");
  const [selectedLLMVoice, setSelectedLLMVoice] = useState("");
  const [selectedLLMChat, setSelectedLLMChat] = useState("");
  const [selectedTTS, setSelectedTTS] = useState("");
  const [selectedASR, setSelectedASR] = useState("");
  const [activeTab, setActiveTab] = useState("voice"); // New state for active tab
  const [billingPeriod, setBillingPeriod] = useState("Month");
  
  // Calculation state
  const [calculations, setCalculations] = useState({
    voiceCreditsPerMinute: 0,
    voiceTotalCredits: 0,
    chatCreditsPerMessage: 0,
    chatTotalCredits: 0,
    totalCredits: 0,
    recommendedPlan: "BUSINESS"
  });

  // Fetch data from API
  useEffect(() => {
    async function fetchCreditsData() {
      setLoading(true);
      setError(null);
      try {
        const res = await fetch('https://realtime-api.voiceflow.com/v1alpha1/estimation/config');
        if (!res.ok) throw new Error('Failed to fetch pricing data');
        const data = await res.json();
        
        // Transform usage data to match existing structure
        const transformedCreditsData = {
          orchestration: [],
          llm: [],
          asr: [],
          tts: []
        };

        // Group usage by category and transform to existing format
        for (const item of data.config.usage) {
          const transformedItem = {
            vendor: getVendorDisplayName(item.vendor),
            item: getItemDisplayName(item.item, item.vendor, item.category),
            unitType: getUnitType(item.unit),
            unitMultiplier: getUnitMultiplier(item.unit),
            unit: getUnitDisplayName(item.unit),
            creditsPerUnit: item.value,
            creditsPerBaseUnit: item.value / getUnitMultiplier(item.unit)
          };

          if (transformedCreditsData[item.category]) {
            transformedCreditsData[item.category].push(transformedItem);
          }
        }

        // Add orchestration items that match the existing structure
        transformedCreditsData.orchestration = [
          { 
            vendor: "Voiceflow", 
            item: "Chat AI Hosting", 
            unitType: "messages",
            unitMultiplier: 1,
            unit: "Agent Sent Message",
            creditsPerUnit: data.config.usage.find(u => u.item === 'interaction')?.value || 1,
            creditsPerBaseUnit: data.config.usage.find(u => u.item === 'interaction')?.value || 1
          },
          { 
            vendor: "Voiceflow", 
            item: "Voice AI Hosting", 
            unitType: "minutes",
            unitMultiplier: 1,
            unit: "Minutes of Voice AI",
            creditsPerUnit: data.config.usage.find(u => u.item === 'audio')?.value || 10,
            creditsPerBaseUnit: data.config.usage.find(u => u.item === 'audio')?.value || 10
          }
        ];

        // Transform plans data
        const transformedPlans = [];
        for (const planGroup of data.config.allotment) {
          for (const plan of planGroup) {
            transformedPlans.push({
                             name: plan.item.includes('pro') ? "PRO" : plan.item.includes('team') ? "BUSINESS" : "ENTERPRISE",
              period: plan.period === 'month' ? "Month" : "Year",
              credits: plan.credits,
              price: plan.price / 100, // Convert from cents to dollars
              unitCost: (plan.price / 100) / plan.credits
            });
          }
        }

                 setCreditsData(transformedCreditsData);
         setPlans(transformedPlans);
         
         // Set default selections once data is loaded
         if (!selectedLLMVoice && transformedCreditsData.llm.length > 0) {
           const defaultLLM = transformedCreditsData.llm.find(llm => llm.item === 'gpt-4.1-2025-04-14') || transformedCreditsData.llm[0];
           setSelectedLLMVoice(defaultLLM.item);
         }
         if (!selectedLLMChat && transformedCreditsData.llm.length > 0) {
           const defaultChatLLM = transformedCreditsData.llm.find(llm => llm.item === 'gpt-4o-mini-2024-07-18') || transformedCreditsData.llm[0];
           setSelectedLLMChat(defaultChatLLM.item);
         }
         if (!selectedTTS && transformedCreditsData.tts.length > 0) {
           const defaultTTS = transformedCreditsData.tts.find(tts => tts.item === 'Eleven Labs - TTS - Flash/Turbo') || transformedCreditsData.tts[0];
           setSelectedTTS(defaultTTS.item);
         }
         if (!selectedASR && transformedCreditsData.asr.length > 0) {
           const defaultASR = transformedCreditsData.asr.find(asr => asr.item === 'Deepgram - STT') || transformedCreditsData.asr[0];
           setSelectedASR(defaultASR.item);
         }
       } catch (e) {
         setError(e.message);
       } finally {
         setLoading(false);
       }
     }
     fetchCreditsData();
   }, [selectedLLMVoice, selectedLLMChat, selectedTTS, selectedASR]);

  // Helper functions for data transformation
  const getVendorDisplayName = (vendor) => {
    const vendorMap = {
      'open-ai': 'Open AI',
      'anthropic': 'Anthropic',
      'google': 'Google',
      'groq': 'Groq',
      'deepgram': 'Deepgram',
      'rimelabs': 'Rime Labs',
      'cartesia': 'Cartesia',
      'elevenlabs': 'Eleven Labs',
      'assemblyai': 'Assemblyai',
      'voiceflow': 'Voiceflow'
    };
    return vendorMap[vendor] || vendor;
  };

  const getItemDisplayName = (item, vendor, category) => {
    // For TTS items
    if (category === 'tts') {
      if (vendor === 'deepgram') return 'Deepgram - TTS';
      if (vendor === 'rimelabs') return 'Rime Labs - TTS';
      if (vendor === 'cartesia') return 'Cartesia - TTS';
      if (vendor === 'elevenlabs') {
        if (item.includes('flash')) return 'Eleven Labs - TTS - Flash/Turbo';
        if (item.includes('turbo')) return 'Eleven Labs - TTS - Flash/Turbo';
        if (item.includes('multilingual')) return 'Eleven Labs - TTS - Multilingual';
        return 'Eleven Labs - TTS';
      }
    }
    
    // For ASR items
    if (category === 'asr') {
      if (vendor === 'deepgram') return 'Deepgram - STT';
      if (vendor === 'cartesia') return 'Cartesia - STT';
      if (vendor === 'assemblyai') return 'Assemblyai - STT';
    }

    // For LLM items, return the item name directly
    return item;
  };

  const getUnitType = (unit) => {
    if (unit.includes('tokens')) return 'tokens';
    if (unit.includes('characters')) return 'characters';
    if (unit.includes('minute')) return 'minutes';
    return 'messages';
  };

  const getUnitMultiplier = (unit) => {
    if (unit.includes('1000')) return 1000;
    return 1;
  };

  const getUnitDisplayName = (unit) => {
    if (unit.includes('1000 LLM tokens')) return '1K Tokens';
    if (unit.includes('1000 characters')) return '1K Characters';
    if (unit.includes('minute')) return 'Minutes';
    return 'Messages';
  };

  // Helper function to find service by item name
  const findService = (category, itemName) => {
    return creditsData[category].find(service => service.item === itemName);
  };

  // Calculate credits
  useEffect(() => {
    if (!creditsData || !plans.length) return;

    // Voice AI calculations
    const voiceHosting = 10; // Fixed Voiceflow Voice AI Hosting credits per minute
    const llmVoice = findService('llm', selectedLLMVoice);
    const tts = findService('tts', selectedTTS);
    const asr = findService('asr', selectedASR);
    
    // Voice AI assumptions from the sheet
    const ttsUtilizationRate = 0.70; // 70%
    const ttsCharsPerMinute = 700;
    const asrUtilizationRate = 0.85; // 85%
    const llmTokensPerMinute = 600;
    
    // Calculate effective usage
    const effectiveTtsCharsPerMinute = ttsCharsPerMinute * ttsUtilizationRate; // 490 chars
    
    // Calculate credits using the new structure
    const llmCreditsPerVoiceMinute = llmVoice ? (llmVoice.creditsPerBaseUnit * llmTokensPerMinute) : 0;
    const ttsCreditsPerVoiceMinute = tts ? (tts.creditsPerBaseUnit * effectiveTtsCharsPerMinute) : 0;
    const asrCreditsPerVoiceMinute = asr ? (asr.creditsPerBaseUnit * asrUtilizationRate) : 0;
    
    const voiceCreditsPerMinute = voiceHosting + llmCreditsPerVoiceMinute + ttsCreditsPerVoiceMinute + asrCreditsPerVoiceMinute;
    const voiceTotalCredits = voiceCreditsPerMinute * (voiceMinutes || 0);

    // Chat AI calculations
    const chatHosting = 1; // Fixed Voiceflow Chat AI Hosting credits per message
    const llmChat = findService('llm', selectedLLMChat);
    
    // Chat AI assumptions from the sheet
    const llmTokensPerMessage = 600;
    
    // Calculate credits using the new structure
    const llmCreditsPerChatMessage = llmChat ? (llmChat.creditsPerBaseUnit * llmTokensPerMessage) : 0;
    const chatCreditsPerMessage = chatHosting + llmCreditsPerChatMessage;
    const chatTotalCredits = chatCreditsPerMessage * (chatMessages || 0);

    const totalCredits = voiceTotalCredits + chatTotalCredits;
    
    // Determine recommended plan
    const filteredPlans = plans.filter(plan => plan.period === billingPeriod);
    const recommended = filteredPlans.find(plan => plan.credits >= totalCredits);
    const recommendedPlan = recommended || null;

    // Find the recommended plan object for cost calculation
    const planUnitCost = recommended ? recommended.unitCost : (billingPeriod === 'Year' ? 0.0054 : 0.005);
    const estimatedCost = billingPeriod === 'Year'
      ? totalCredits * planUnitCost
      : totalCredits * planUnitCost;
    const estimatedCostLabel = billingPeriod === 'Year' ? 'Estimated Cost/Year' : 'Estimated Cost/Month';

    setCalculations({
      voiceCreditsPerMinute,
      voiceTotalCredits,
      chatCreditsPerMessage,
      chatTotalCredits,
      totalCredits,
      recommendedPlan,
      estimatedCost,
      estimatedCostLabel,
      // Add individual service breakdowns for display
      voiceBreakdown: {
        hosting: voiceHosting,
        llm: llmCreditsPerVoiceMinute,
        tts: ttsCreditsPerVoiceMinute,
        asr: asrCreditsPerVoiceMinute
      },
      chatBreakdown: {
        hosting: chatHosting,
        llm: llmCreditsPerChatMessage
      }
    });
  }, [voiceMinutes, chatMessages, selectedLLMVoice, selectedLLMChat, selectedTTS, selectedASR, billingPeriod, creditsData, plans]);

  // Helper function to format numbers
  const formatNumber = (num) => {
    return new Intl.NumberFormat().format(Math.round(num));
  };

  // Helper function to format currency
  const formatCurrency = (num) => {
    if (num < 0.01) {
      return new Intl.NumberFormat('en-US', { 
        style: 'currency', 
        currency: 'USD',
        minimumFractionDigits: 4,
        maximumFractionDigits: 4 
      }).format(num);
    }
    return new Intl.NumberFormat('en-US', { 
      style: 'currency', 
      currency: 'USD' 
    }).format(num);
  };

  // Helper to get recommended plan for a given total credits and billing period
  const getRecommendedPlan = (totalCredits, billingPeriod) => {
    if (!plans.length) return null;
    const filteredPlans = plans.filter(plan => plan.period === billingPeriod);
    const recommended = filteredPlans.find(plan => plan.credits >= totalCredits);
    return recommended || null;
  };

  return (
    <div className="max-w-6xl mx-auto">
      {/* Calculator Selector */}
      <div className="mb-8">
        <div className="flex flex-wrap items-center gap-4 text-left">
          <span className="text-base font-normal text-gray-900 dark:text-white whitespace-nowrap">I'm building a</span>
          <div className="flex gap-3 flex-wrap">
            <button
              onClick={() => setActiveTab("voice")}
              className={`px-5 py-2 rounded-lg border transition-all duration-200 flex items-center gap-2 text-base font-normal
                ${activeTab === "voice"
                  ? "bg-blue-50 dark:bg-blue-900/20 border-blue-500 text-blue-800 dark:text-blue-200"
                  : "bg-white dark:bg-gray-800 border-gray-300 dark:border-gray-700 text-gray-700 dark:text-gray-200 hover:border-blue-300"}
              `}
            >
              <i className="fas fa-microphone text-lg"></i>
              Voice agent
            </button>
            <button
              onClick={() => setActiveTab("chat")}
              className={`px-5 py-2 rounded-lg border transition-all duration-200 flex items-center gap-2 text-base font-normal
                ${activeTab === "chat"
                  ? "bg-blue-50 dark:bg-blue-900/20 border-blue-500 text-blue-800 dark:text-blue-200"
                  : "bg-white dark:bg-gray-800 border-gray-300 dark:border-gray-700 text-gray-700 dark:text-gray-200 hover:border-blue-300"}
              `}
            >
              <i className="fas fa-comments text-lg"></i>
              Chat agent
            </button>
          </div>
        </div>
        <br />
        {/* Billing Period Selector */}
        <div className="flex flex-wrap items-center gap-4 text-left">
          <span className="text-base font-normal text-gray-900 dark:text-white whitespace-nowrap">I'm billed</span>
          <div className="flex gap-3 flex-wrap">
            <button
              onClick={() => setBillingPeriod("Month")}
              className={`px-5 py-2 rounded-lg border transition-all duration-200 flex items-center gap-2 text-base font-normal
                ${billingPeriod === "Month"
                  ? "bg-blue-50 dark:bg-blue-900/20 border-blue-500 text-blue-800 dark:text-blue-200"
                  : "bg-white dark:bg-gray-800 border-gray-300 dark:border-gray-700 text-gray-700 dark:text-gray-200 hover:border-blue-300"}
              `}
            >
              Monthly
            </button>
            <button
              onClick={() => setBillingPeriod("Year")}
              className={`px-5 py-2 rounded-lg border transition-all duration-200 flex items-center gap-2 text-base font-normal
                ${billingPeriod === "Year"
                  ? "bg-blue-50 dark:bg-blue-900/20 border-blue-500 text-blue-800 dark:text-blue-200"
                  : "bg-white dark:bg-gray-800 border-gray-300 dark:border-gray-700 text-gray-700 dark:text-gray-200 hover:border-blue-300"}
              `}
            >
              Annually (10% discount)
            </button>
          </div>
        </div>
      </div>

      {/* Calculator Tabs */}
      <div className="mb-8">
        {/* Tab Content */}
        <div className="p-8 bg-white dark:bg-gray-800 border border-gray-700 rounded-xl">
          {loading && <p>Loading pricing data...</p>}
          {error && <p style={{ color: 'red' }}>{error}</p>}
          {creditsData && (
            <>
              {activeTab === "voice" && (
                <div className="p-0">
                  {/* Voice Minutes Input (label/placeholder changes with billingPeriod) */}
                  <div className="mb-8">
                    <label className="block text-base font-semibold text-gray-700 dark:text-gray-300 mb-3">
                      {billingPeriod === 'Year' ? 'Call minutes per year' : 'Call minutes per month'}
                    </label>
                    <input 
                      type="text" 
                      value={voiceMinutesInput}
                      onChange={(e) => {
                        const value = e.target.value;
                        setVoiceMinutesInput(value);
                        setVoiceMinutes(value === '' ? 0 : Number(value));
                      }}
                      className="w-full px-4 py-3 border-2 border-gray-200 dark:border-gray-600 rounded-xl bg-white dark:bg-gray-700 text-gray-900 dark:text-white focus:ring-4 focus:ring-blue-500/20 focus:border-blue-500 transition-all duration-200 text-lg"
                      placeholder={billingPeriod === 'Year' ? 'Enter call minutes per year' : 'Enter call minutes per month'}
                    />
                    <div className="mt-2 text-sm font-medium text-gray-500 dark:text-gray-400">
                      Credits per minute: 10.00
                    </div>
                  </div>
                  {/* Service Configuration */}
                  <div className="space-y-6">
                    <div>
                      <label className="block text-base font-semibold text-gray-700 dark:text-gray-300 mb-3">LLM model</label>
                      <div className="relative">
                        <select 
                          value={selectedLLMVoice}
                          onChange={(e) => setSelectedLLMVoice(e.target.value)}
                          className="appearance-none w-full px-4 py-3 border-2 border-gray-200 dark:border-gray-600 rounded-xl bg-white dark:bg-gray-700 text-gray-900 dark:text-white focus:ring-4 focus:ring-orange-500/20 focus:border-orange-500 transition-all duration-200"
                        >
                          {creditsData.llm.map((llm) => (
                            <option key={llm.item} value={llm.item}>
                              {llm.vendor} - {llm.item}
                            </option>
                          ))}
                        </select>
                        <span className="pointer-events-none absolute right-4 top-1/2 transform -translate-y-1/2 text-gray-400">▼</span>
                      </div>
                      <div className="mt-2 text-sm font-medium text-gray-500 dark:text-gray-400">
                        Credits per minute: {calculations.voiceBreakdown?.llm?.toFixed(2) || '0.00'}
                      </div>
                    </div>

                    <div>
                      <label className="block text-base font-semibold text-gray-700 dark:text-gray-300 mb-3">Text-to-speech</label>
                      <div className="relative">
                        <select 
                          value={selectedTTS}
                          onChange={(e) => setSelectedTTS(e.target.value)}
                          className="appearance-none w-full px-4 py-3 border-2 border-gray-200 dark:border-gray-600 rounded-xl bg-white dark:bg-gray-700 text-gray-900 dark:text-white focus:ring-4 focus:ring-orange-500/20 focus:border-orange-500 transition-all duration-200"
                        >
                          {creditsData.tts.map((tts) => (
                            <option key={tts.item} value={tts.item}>
                              {tts.item}
                            </option>
                          ))}
                        </select>
                        <span className="pointer-events-none absolute right-4 top-1/2 transform -translate-y-1/2 text-gray-400">▼</span>
                      </div>
                      <div className="mt-2 text-sm font-medium text-gray-500 dark:text-gray-400">
                        Credits per minute: {calculations.voiceBreakdown?.tts?.toFixed(2) || '0.00'}
                      </div>
                    </div>

                    <div>
                      <label className="block text-base font-semibold text-gray-700 dark:text-gray-300 mb-3">Speech-to-text</label>
                      <div className="relative">
                        <select 
                          value={selectedASR}
                          onChange={(e) => setSelectedASR(e.target.value)}
                          className="appearance-none w-full px-4 py-3 border-2 border-gray-200 dark:border-gray-600 rounded-xl bg-white dark:bg-gray-700 text-gray-900 dark:text-white focus:ring-4 focus:ring-orange-500/20 focus:border-orange-500 transition-all duration-200"
                        >
                          {creditsData.asr.map((asr) => (
                            <option key={asr.item} value={asr.item}>
                              {asr.item}
                            </option>
                          ))}
                        </select>
                        <span className="pointer-events-none absolute right-4 top-1/2 transform -translate-y-1/2 text-gray-400">▼</span>
                      </div>
                      <div className="mt-2 text-sm font-medium text-gray-500 dark:text-gray-400">
                        Credits per minute: {calculations.voiceBreakdown?.asr?.toFixed(2) || '0.00'}
                      </div>
                    </div>
                  </div>
                </div>
              )}

              {activeTab === "chat" && (
                <div className="p-0">
                  {/* Chat Messages Input (label/placeholder changes with billingPeriod) */}
                  <div className="mb-8">
                    <label className="block text-base font-semibold text-gray-700 dark:text-gray-300 mb-3">
                      {billingPeriod === 'Year' ? 'Interactions per year' : 'Interactions per month'}
                    </label>
                    <input 
                      type="text" 
                      value={chatMessagesInput}
                      onChange={(e) => {
                        const value = e.target.value;
                        setChatMessagesInput(value);
                        setChatMessages(value === '' ? 0 : Number(value));
                      }}
                      className="w-full px-4 py-3 border-2 border-gray-200 dark:border-gray-600 rounded-xl bg-white dark:bg-gray-700 text-gray-900 dark:text-white focus:ring-4 focus:ring-emerald-500/20 focus:border-emerald-500 transition-all duration-200 text-lg"
                      placeholder={billingPeriod === 'Year' ? 'Enter messages per year' : 'Enter messages per month'}
                    />
                    <div className="mt-2 text-sm font-medium text-gray-500 dark:text-gray-400">
                      Credits per interaction: 1.00
                    </div>
                  </div>
                  {/* Service Configuration */}
                  <div className="space-y-6">
                    <div>
                      <label className="block text-base font-semibold text-gray-700 dark:text-gray-300 mb-3">LLM model</label>
                      <div className="relative">
                        <select 
                          value={selectedLLMChat}
                          onChange={(e) => setSelectedLLMChat(e.target.value)}
                          className="appearance-none w-full px-4 py-3 border-2 border-gray-200 dark:border-gray-600 rounded-xl bg-white dark:bg-gray-700 text-gray-900 dark:text-white focus:ring-4 focus:ring-orange-500/20 focus:border-orange-500 transition-all duration-200"
                        >
                          {creditsData.llm.map((llm) => (
                            <option key={llm.item} value={llm.item}>
                              {llm.vendor} - {llm.item}
                            </option>
                          ))}
                        </select>
                        <span className="pointer-events-none absolute right-4 top-1/2 transform -translate-y-1/2 text-gray-400">▼</span>
                      </div>
                      <div className="mt-2 text-sm font-medium text-gray-500 dark:text-gray-400">
                        Credits per message: {calculations.chatBreakdown?.llm?.toFixed(2) || '0.00'}
                      </div>
                    </div>
                  </div>
                </div>
              )}
            </>
          )}
        </div>
      </div>

      {/* Voice summary box OUTSIDE the container */}
      {activeTab === "voice" && !loading && creditsData && (
        <div className="p-8 bg-white dark:bg-gray-800 border border-gray-700 rounded-xl">
          <div className="flex flex-col gap-2">
            <div className="flex justify-between items-center">
              <div className="text-base font-medium text-gray-700 dark:text-gray-300">Credits per minute</div>
              <div className="text-2xl font-bold text-blue-700 dark:text-blue-300">{calculations.voiceCreditsPerMinute.toFixed(2)}</div>
            </div>
            <div className="flex justify-between items-center">
              <div className="text-base font-medium text-gray-700 dark:text-gray-300">Cost per minute</div>
              <div className="text-2xl font-bold text-emerald-600 dark:text-emerald-400">{formatCurrency(calculations.voiceCreditsPerMinute * (billingPeriod === 'Year' ? 0.0045 : 0.005))}</div>
            </div>
            <div className="flex justify-between items-center">
              <div className="text-base font-medium text-gray-700 dark:text-gray-300">Credits required</div>
              <div className="text-2xl font-bold text-gray-900 dark:text-white">{formatNumber(calculations.voiceTotalCredits)}</div>
            </div>
            <div className="flex justify-between items-center">
              <div className="text-base font-medium text-gray-700 dark:text-gray-300">Recommended plan</div>
              {(() => {
                const plan = getRecommendedPlan(calculations.voiceTotalCredits, billingPeriod);
                return plan ? (
                  <div className="text-base font-bold text-blue-700 dark:text-blue-300 text-right">
                    {plan.name} ({plan.credits.toLocaleString()} credits/{plan.period.toLowerCase()})
                  </div>
                ) : (
                  <div className="text-base font-bold text-blue-700 dark:text-blue-300 text-right">Enterprise</div>
                );
              })()}
            </div>
            <div className="flex justify-between items-center">
              <div className="text-base font-medium text-gray-700 dark:text-gray-300">{billingPeriod === 'Year' ? 'Annual cost' : 'Monthly cost'}</div>
              {(() => {
                const plan = getRecommendedPlan(calculations.voiceTotalCredits, billingPeriod);
                return plan ? (
                  <div className="text-base font-bold text-blue-700 dark:text-blue-300 text-right">{formatCurrency(plan.price)} <span className="text-gray-500">{billingPeriod === 'Year' ? 'per year' : 'per month'}</span></div>
                ) : (
                  <div className="text-base font-bold text-blue-700 dark:text-blue-300 text-right">Contact sales</div>
                );
              })()}
            </div>
          </div>
        </div>
      )}

      {/* Chat summary box OUTSIDE the container */}
      {activeTab === "chat" && !loading && creditsData && (
        <div className="p-8 bg-white dark:bg-gray-800 border border-gray-700 rounded-xl">
          <div className="flex flex-col gap-2">
            <div className="flex justify-between items-center">
              <div className="text-base font-medium text-gray-700 dark:text-gray-300">Credits per Message</div>
              <div className="text-2xl font-bold text-emerald-700 dark:text-emerald-300">{calculations.chatCreditsPerMessage.toFixed(2)}</div>
            </div>
            <div className="flex justify-between items-center">
              <div className="text-base font-medium text-gray-700 dark:text-gray-300">Cost per Message</div>
              <div className="text-2xl font-bold text-emerald-600 dark:text-emerald-400">{formatCurrency(calculations.chatCreditsPerMessage * (billingPeriod === 'Year' ? 0.0045 : 0.005))}</div>
            </div>
            <div className="flex justify-between items-center">
              <div className="text-base font-medium text-gray-700 dark:text-gray-300">Credits Required</div>
              <div className="text-2xl font-bold text-gray-900 dark:text-white">{formatNumber(calculations.chatTotalCredits)}</div>
            </div>
            <div className="flex justify-between items-center">
              <div className="text-base font-medium text-gray-700 dark:text-gray-300">Recommended Plan</div>
              {(() => {
                const plan = getRecommendedPlan(calculations.chatTotalCredits, billingPeriod);
                return plan ? (
                  <div className="text-base font-bold text-blue-700 dark:text-blue-300 text-right">
                    {plan.name} ({plan.credits.toLocaleString()} credits/{plan.period.toLowerCase()})
                  </div>
                ) : (
                  <div className="text-base font-bold text-blue-700 dark:text-blue-300 text-right">Enterprise</div>
                );
              })()}
            </div>
            <div className="flex justify-between items-center">
              <div className="text-base font-medium text-gray-700 dark:text-gray-300">{billingPeriod === 'Year' ? 'Annual Cost' : 'Monthly Cost'}</div>
              {(() => {
                const plan = getRecommendedPlan(calculations.chatTotalCredits, billingPeriod);
                return plan ? (
                  <div className="text-base font-bold text-blue-700 dark:text-blue-300 text-right">{formatCurrency(plan.price)} <span className="text-gray-500">{billingPeriod === 'Year' ? 'per year' : 'per month'}</span></div>
                ) : (
                  <div className="text-base font-bold text-blue-700 dark:text-blue-300 text-right">Contact Sales</div>
                );
              })()}
            </div>
          </div>
        </div>
      )}
    </div>
