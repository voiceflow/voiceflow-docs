  const [selectedLLMVoice, setSelectedLLMVoice] = useState("gpt-4.1-2025-04-14");

  const [activeTab, setActiveTab] = useState("voice"); // New state for active tab
  
  // Calculation state

    voiceCreditsPerMinute: 0,
    voiceTotalCredits: 0,
    chatCreditsPerMessage: 0,

  // Helper function to find service by item name
  const findService = (category, itemName) => {

  // Calculate credits

    // Voice AI calculations
    const voiceHosting = 10; // Fixed Voiceflow Voice AI Hosting credits per minute
    const llmVoice = findService('llm', selectedLLMVoice);
    const tts = findService('tts', selectedTTS);
    const asr = findService('asr', selectedASR);

    const ttsUtilizationRate = 0.70; // 70%

    // Calculate effective usage
    const effectiveTtsCharsPerMinute = ttsCharsPerMinute * ttsUtilizationRate; // 490 chars

    const llmCreditsPerVoiceMinute = llmVoice ? (llmVoice.creditsPerUnit * effectiveLlmTokensPerMinute / 1000) : 0;

    const voiceTotalCredits = voiceCreditsPerMinute * voiceMinutes;

    // Chat AI calculations

    const chatTotalCredits = chatCreditsPerMessage * chatMessages;

    const totalCredits = voiceTotalCredits + chatTotalCredits;
    
    // Determine recommended plan
    const recommendedPlan = plans.find(plan => plan.creditsPerMonth >= totalCredits)?.name || "ENTERPRISE";

      voiceCreditsPerMinute,
      voiceTotalCredits,
      chatCreditsPerMessage,
      chatTotalCredits,
      totalCredits,
      recommendedPlan,

        hosting: voiceHosting,
        llm: llmCreditsPerVoiceMinute,
        tts: ttsCreditsPerVoiceMinute,
        asr: asrCreditsPerVoiceMinute
      },
      chatBreakdown: {
        hosting: chatHosting,
        llm: llmCreditsPerChatMessage
      }
    });
  }, [voiceMinutes, chatMessages, selectedLLMVoice, selectedLLMChat, selectedTTS, selectedASR]);

  };

  const formatCurrency = (num) => {
    return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(num);

                    <span className="text-2xl font-bold text-gray-900 dark:text-white">{formatNumber(calculations.chatTotalCredits)}</span>
                  </div>
                  <div className="flex justify-between items-center">
                    <span className="text-base font-medium text-gray-700 dark:text-gray-300">Cost per Message</span>

                    </div>
                  </div>

            )}
